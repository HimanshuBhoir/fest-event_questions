void count_odd_even(int arr[], int size) {
    int odd = 0;
    int even = 0;
    for (int i = 0; i < size; i++) {
        if (arr[i] % 2 == 0) {
            even += 1;
        } else {
            odd += 1;
        }
    }
    printf("oddCount: %d\n", odd);
    printf("evenCount: %d\n", even);
}
