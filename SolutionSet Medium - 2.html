    public class Solution {
    public static void main(String[] args) {
        int number = 13;
        boolean isPrime = checkPrime.test(number);
        System.out.println(isPrime ? "Prime" : "Not Prime");
    }

    private static IntPredicate checkPrime = number ->
            IntStream.rangeClosed(2, (int) Math.sqrt(number))
                     .allMatch(i -> number % i != 0) && number > 1;
}

    // Fibonacci series
#include <iostream>
#include <vector>
std::vector<int> fibonacciSeries(int numTerms) {
    std::vector<int> fibSeries{0, 1}; 
    for (int i = 2; i < numTerms; i++) {
        int nextFib = fibSeries[i - 1] + fibSeries[i - 2];
        fibSeries.push_back(nextFib);
    }
    return fibSeries;
}
int main() {
    int numTerms;
    std::cout << "Enter the number of terms for Fibonacci series: ";
    std::cin >> numTerms;
    std::vector<int> fibSeries = fibonacciSeries(numTerms);
    std::cout << "Fibonacci series:";
    for (int i = 0; i < fibSeries.size(); i++) {
        std::cout << " " << fibSeries[i];
    }
    std::cout << std::endl;

    return 0;
}
