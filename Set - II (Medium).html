std::vector<int> fibonacciSeries(int numTerms) {
    std::vector<int> fibSeries{0, 1}; 
    for (int i = 2; i < numTerms; i++) {
        int nextFib = fibSeries[i - 1] + fibSeries[i - 2];
        fibSeries.push_back(nextFib);
    }
    return fibSeries;
}

int main() {
    int numTerms;
    var numTerms = parseInt(prompt("Enter the number of terms for Fibonacci series:"));
    std::vector<int> fibSeries = fibonacciSeries(numTerms);
    std::cout << "Fibonacci series:";
    for (int i = 0; i < fibSeries.size(); i++) {
        std::cout << " " << fibSeries[i];
    }
    std::cout << std::endl;

    return 0;
}
